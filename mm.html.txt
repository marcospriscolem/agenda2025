  <script>
    function calcularRaizes() {
      const a = parseFloat(document.getElementById('a').value);
      const b = parseFloat(document.getElementById('b').value);
      const c = parseFloat(document.getElementById('c').value);
      const resultado = document.getElementById('resultado');

      if (isNaN(a) || isNaN(b) || isNaN(c)) {
        resultado.innerHTML = "Por favor, insira valores válidos para a, b e c.";
        return;
      }

      if (a === 0) {
        resultado.innerHTML = "Isso não é uma equação do 2º grau (a ≠ 0).";
        return;
      }

      const delta = b * b - 4 * a * c;

      if (delta > 0) {
        const x1 = (-b + Math.sqrt(delta)) / (2 * a);
        const x2 = (-b - Math.sqrt(delta)) / (2 * a);
        resultado.innerHTML = `Duas raízes reais: x₁ = ${x1.toFixed(2)}, x₂ = ${x2.toFixed(2)}`;
      } else if (delta === 0) {
        const x = -b / (2 * a);
        resultado.innerHTML = `Uma raiz real: x = ${x.toFixed(2)}`;
      } else {
        const parteReal = (-b / (2 * a)).toFixed(2);
        const parteImaginaria = (Math.sqrt(-delta) / (2 * a)).toFixed(2);
        resultado.innerHTML = `Duas raízes complexas: x₁ = ${parteReal} + ${parteImaginaria}i, x₂ = ${parteReal} - ${parteImaginaria}i`;
      }
    }
  </script>
